SELECT ?surfaceTypeName (COUNT(*) AS ?binCount)
    WHERE {
    ?temp1  rdf-schema:binNo ?binNo.
	?temp1  rdf-schema:street ?streetName
	BIND(IRI(CONCAT("http://www.example.org/bin-features/", ENCODE_FOR_URI(?binNo))) AS ?binFeaturesURI)
    ?binFeaturesURI  foaf:advertSpace ?hasAdvertSpace.
	BIND(IRI(CONCAT("http://www.example.org/street/", ENCODE_FOR_URI(?streetName))) AS ?streetURI)
    ?streetURI  rdf-schema:surfaceTypeCode ?surfaceTypeCodeURI.
    ?surfaceTypeCodeURI  foaf:surfaceTypeName ?surfaceTypeName.
    FILTER (regex(str(?temp1), "bin") && regex(str(?hasAdvertSpace), "TRUE"))
    }
    GROUP BY ?surfaceTypeName
    ORDER BY ?binCount
